def create_data(sss,para,i=0):
    if sss.shape[1] == 1:
        train = sss.copy() #series
    else:
        train = sss.iloc[:,i:(i+1)].copy() #dataframe
    train.columns = ['main']
    for i in para:
        train[('ma'+str(i))] = train['main'].rolling(i).mean()
    return train

def trend_follow_n_sig(train,diff):
    '''
    ;train;diff 
    '''
    sig = np.array([0,0]) #
    diff = signal(diff) #-1 1
    
    for i in range(2,len(train)):  # 
        if (diff[i-2] in [0,-1]) and (diff[i-1] == 1): #
            sig = np.append(sig,1) #
        elif (diff[i-2] == 1) and (diff[i-1] == -1): #
            sig = np.append(sig,-1) #        
        elif np.isnan(diff[i-2]) or np.isnan(diff[i-1]): #
            sig = np.append(sig,0) #0
        elif (diff[i-2] == diff[i-1]):  # 
            sig = np.append(sig,sig[-1])
            
    return sig

def trend_follow_l_sig(train,diff):
    '''
    ;train;diff 
    '''
    sig = np.array([0, 0])
    diff = signal(diff) # 1 -1 
    
    for i in range(2,len(train)):  # 
        if (diff[i-2] in [0,-1]) and (diff[i-1] == 1): #
            sig = np.append(sig, 1)
        elif (diff[i-2] == 1) and (diff[i-1] == -1): #
            sig = np.append(sig, 0)        
        elif np.isnan(diff[i-2]) or np.isnan(diff[i-1]): #
            sig = np.append(sig, 0)                
        elif (diff[i-2] == diff[i-1]):  # 
            sig = np.append(sig, sig[-1])
            
    return sig

def trend_follow_k_delta_sig(train,diff,diff1,kl,ks,delta): 
    '''
    ;train
    diff ;diff1
    k;
    '''
    vol_l = (train['main'] / train['main'].rolling(50).min() - 1) #50
    vol_s = (train['main'] / train['main'].rolling(50).max() - 1) #50
    vol_l.fillna(0,inplace=True)
    vol_s.fillna(0,inplace=True) 

    sig = np.array([0, 0]) #
    lng = 0  # 
    sht = 0  # 
    origin = 0 #()
    diff = signal_s(diff,delta) #-1 0 1
    diff1 = signal(diff1) #-1 1
    
    for i in range(2,len(train)):  # 
        if (diff[i-2] in [-1,1]) and (diff[i-1] == 0): #()
            sig = np.append(sig, 0) #0
            sht, lng = 0, 0 #
        elif (diff[i-2] in [0,-1]) and (diff[i-1] == 1): #
            sig = np.append(sig, 1) #1
            sht, lng = 0, 1 #
        elif (diff[i-2] in [0,1]) and (diff[i-1] == -1): #
            sig = np.append(sig, -1) #-1
            lng, sht = 0, 1 #
        elif np.isnan(diff[i-2]) or np.isnan(diff[i-1]): #
            sig = np.append(sig, 0) #na                
        elif (diff[i-2] == diff[i-1]):  # +
            if (vol_l[i-1] < kl) and (vol_s[i-1] > ks): #
                if lng + sht !=0:
                    sig = np.append(sig, sig[-1]) #
                elif lng + sht == 0:
                    if diff[i-1] == 1:
                        sig = np.append(sig, 1) #
                        lng = 1
                    elif diff[i-1] == -1:
                        sig = np.append(sig, -1) #
                        sht = 1
                    else:
                        sig = np.append(sig, 0) #            
            elif (vol_l[i-1] >= kl) or (vol_s[i-1] <= ks): #
                if (diff1[i-2] * diff1[i-1]) == -1: #
                    sig = np.append(sig, 0) #
                    sht, lng = 0, 0 #
                elif (diff1[i-2] == diff1[i-1]):
                    if lng + sht !=0:
                        sig = np.append(sig, sig[-1]) #
                    elif lng + sht == 0: #
                        if diff1[i-1] == 1 and diff[i-1] == 1: #
                            sig = np.append(sig, 1)
                            lng = 1
                        elif diff1[i-1] == -1 and diff[i-1] == -1: #
                            sig = np.append(sig, -1)
                            sht = 1
                        else: #
                            sig = np.append(sig, 0) #

    return sig
